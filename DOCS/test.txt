const API_BASE = import.meta.env.VITE_API_BASE ?? "/api";

function fireToast(title, description, variant = "default") {
  if (typeof window === "undefined") return;
  window.dispatchEvent(
    new CustomEvent("app-toast", { detail: { title, description, variant } })
  );
}

export async function apiRequest(path, options = {}) {
  let response;

  try {
    response = await fetch(`${API_BASE}${path}`, {
      headers: {
        "Content-Type": "application/json",
        ...(options.headers ?? {}),
      },
      ...options,
    });
  } catch (networkErr) {
    const msg = "Cannot connect to server. Check your internet connection.";
    fireToast("Connection Error", msg, "destructive");
    throw new Error(msg);
  }

  const contentType = response.headers.get("content-type") ?? "";
  let payload = null;
  if (contentType.includes("application/json")) {
    try {
      payload = await response.json();
    } catch {
      payload = null;
    }
  }

  if (!response.ok) {
    const msg =
      payload?.message ??
      (response.status === 404
        ? "The requested resource was not found."
        : response.status === 400
        ? "Invalid request. Please check your input."
        : response.status >= 500
        ? "A server error occurred. Please try again."
        : `Request failed (${response.status}).`);
    fireToast("Error", msg, "destructive");
    const err = new Error(msg);
    err.status = response.status;
    err.payload = payload;
    throw err;
  }

  if (
    payload?.message &&
    options.method &&
    options.method.toUpperCase() !== "GET"
  ) {
    fireToast("Success", payload.message, "default");
  }

  return payload;
}

export async function apiUpload(path, formData) {
  let response;
  try {
    response = await fetch(`${API_BASE}${path}`, {
      method: "POST",
      body: formData,
    });
  } catch {
    const msg = "Cannot connect to server.";
    fireToast("Connection Error", msg, "destructive");
    throw new Error(msg);
  }

  let payload = null;
  try {
    payload = await response.json();
  } catch {
    payload = null;
  }

  if (!response.ok) {
    const msg = payload?.message ?? "Upload failed.";
    fireToast("Upload Error", msg, "destructive");
    throw new Error(msg);
  }

  return payload;
}

export function formatGenerationId(id) {
  if (!id || typeof id !== "string") return id ?? "—";
  const match = id.match(/^GEN-(\d{4})(\d{2})(\d{2})(\d{2})(\d{2})(\d{2})$/);
  if (!match) return id;
  const [, year, month, day, hour, minute] = match;
  const months = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];
  const h = parseInt(hour, 10);
  const ampm = h >= 12 ? "PM" : "AM";
  const h12 = h % 12 || 12;
  return `Generated on ${parseInt(day, 10)} ${months[parseInt(month, 10) - 1]} ${year} at ${h12}:${minute} ${ampm}`;
}

export function formatDate(iso) {
  if (!iso) return "—";
  try {
    return new Date(iso).toLocaleDateString("en-IN", { day: "numeric", month: "short", year: "numeric" });
  } catch { return "—"; }
}

export function formatDateTime(iso) {
  if (!iso) return "—";
  try {
    return new Date(iso).toLocaleString("en-IN", { day: "numeric", month: "short", year: "numeric", hour: "numeric", minute: "2-digit", hour12: true });
  } catch { return "—"; }
}

export function formatAvailability(str) {
  if (!str) return "—";
  return str.split(",").map(s => s.trim().replace("-", " – ")).join("  ·  ");
}

export function formatSemester(num) {
  if (num === null || num === undefined) return "—";
  return `Semester ${num}`;
}

export function formatProgram(str) {
  if (!str) return "—";
  return str === "UG" ? "Undergraduate" : str === "PG" ? "Postgraduate" : str;
}

export function formatCapacity(num) {
  if (num === null || num === undefined) return "—";
  return `${num} students`;
}

export function formatMaxLectures(num) {
  if (num === null || num === undefined) return "—";
  return `${num} per day`;
}

export function formatBoolean(val) {
  if (val === null || val === undefined) return "—";
  return val ? "Yes" : "No";
}

export function formatCount(num, singular, plural) {
  if (num === null || num === undefined) return "—";
  return `${num} ${num === 1 ? singular : plural}`;
}
